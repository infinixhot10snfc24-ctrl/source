    local serializeTable
    local function serializeValue(value)
        local objectType = typeof(value)
        local primitiveType = type(value)

        if value == nil then
            return 'nil'
        elseif objectType == 'Instance' then
            return getInstancePath(value)
        elseif primitiveType == 'string' then
            return '"' .. value:gsub('\\', '\\\\'):gsub('"', '\\"') .. '"'
        elseif primitiveType == 'boolean' or primitiveType == 'number' then
            return tostring(value)
        elseif primitiveType == 'table' then
            return '{' .. serializeTable(value) .. '}'
        elseif objectType == 'Vector3' then
            return string.format('Vector3.new(%g,%g,%g)', value.X, value.Y, value.Z)
        elseif objectType == 'Vector2' then
            return string.format('Vector2.new(%g,%g)', value.X, value.Y)
        elseif objectType == 'CFrame' then
            return string.format('CFrame.new(%g,%g,%g,%g,%g,%g,%g,%g,%g,%g,%g,%g)', value:GetComponents())
        elseif objectType == 'Color3' then
            return string.format('Color3.new(%g,%g,%g)', value.R, value.G, value.B)
        elseif objectType == 'UDim2' then
            return string.format('UDim2.new(%g,%g,%g,%g)', value.X.Scale, value.X.Offset, value.Y.Scale, value.Y.Offset)
        elseif objectType == 'UDim' then
            return string.format('UDim.new(%g,%g)', value.Scale, value.Offset)
        elseif objectType == 'BrickColor' then
            return string.format('BrickColor.new("%s")', tostring(value))
        elseif objectType == 'EnumItem' then
            return tostring(value)
        elseif objectType == 'buffer' then
            local success, result = pcall(function()
                return string.format('buffer.fromstring("%s")', buffer.tostring(value):gsub('.', function(char) return string.format('\\%d', string.byte(char)) end))
            end)
            return success and result or string.format('buffer(len=%d)', buffer.len(value))
        else
            return tostring(value)
        end
    end

    serializeTable = function(tableValue)
        local keys = {}
        local values = {}
        local count = 0
        for key, val in pairs(tableValue) do
            count = count + 1
            keys[count] = key
            values[count] = val
        end
        local resultString = {}
        local resultIndex = 0
        for i = 1, count do
            local currentKey = keys[i]
            local currentValue = values[i]
            resultIndex = resultIndex + 1
            resultString[resultIndex] = type(currentKey) == 'string' and ('["' .. tostring(currentKey) .. '"] = ') or ('[' .. tostring(currentKey) .. '] = ')
            resultIndex = resultIndex + 1
            resultString[resultIndex] = serializeValue(currentValue)
            if count > 1 and i < count then
                resultIndex = resultIndex + 1
                resultString[resultIndex] = ', '
            end
        end
        return table.concat(resultString)
    end

    local function formatArguments(argsArray)
        local formattedArgs = {}
        local count = 0
        for i = 1, #argsArray do
            count = count + 1
            formattedArgs[count] = serializeValue(argsArray[i])
        end
        return table.concat(formattedArgs, ', ')
    end

    local function normalizeArguments(argsValue)
        if typeof(argsValue) ~= 'table' then return typeof(argsValue) .. ':' .. tostring(argsValue) end
        local sortedArgs = {}
        for key, val in pairs(argsValue) do
            local valType = typeof(val)
            if valType == 'table' then
                sortedArgs[#sortedArgs + 1] = tostring(key) .. '=' .. normalizeArguments(val)
            else
                sortedArgs[#sortedArgs + 1] = tostring(key) .. ':' .. valType
            end
        end
        table.sort(sortedArgs)
        return '{' .. table.concat(sortedArgs, ',') .. '}'
    end

    local function getArgumentsKey(remoteInstance, argsValue)

        infoTitle.Text = 'Info: ' .. remoteInstance.Name

        codeLabel.Text = generatedCode

        ignoreRemoteButton.Text = table.find(ignoredRemotes, getRootParent(remoteInstance)) and 'Unignore Remote' or 'Ignore Remote'

        ignoreArgsButton.Text = (ignoredArgs[remoteKey] and ignoredArgs[remoteKey][argKey]) and 'Unignore Args' or 'Ignore Args'

        playTween(autoFireButton, clickTween, {BackgroundColor3 = colorInteractable})

            buttonConnections[#buttonConnections + 1] = buttonGui.MouseButton1Click:Connect(callbackFunc)

            showFeedback(copyCodeButton)

                    remoteInstance:FireServer(unpack(argsValue))
                else 
                    pcall(remoteInstance.InvokeServer, remoteInstance, unpack(argsValue)) 
                end
            end)
        end)

            end

                return 
            end

                showFeedback(copyDecompiledButton, 'No Script Info')
                return 
            end
            local success, result = pcall(decompile, capturedScript)
            if success and result then 
                setclipboard(result)
                showFeedback(copyDecompiledButton, 'Decompiled!')
            else

    local function captureRemote(isEvent, remoteInstance, ...)
        local rKey, aKey = getArgumentsKey(remoteInstance, {...})
        if spyActive and remoteInstance and not table.find(ignoredRemotes, getRootParent(remoteInstance)) and not (ignoredArgs[rKey] and ignoredArgs[rKey][aKey]) then
            pendingQueue[#pendingQueue + 1] = {isEvent = isEvent, remote = remoteInstance, args = {...}, script = checkcaller and getcallingscript() or nil}
        end
    end

    local function processQueue()
        if not next(pendingQueue) then return end
        local queueBatch = pendingQueue
        pendingQueue = {}
        for _, eventData in ipairs(queueBatch) do
            local indexFound = findRemoteIndex(eventData.remote, eventData.args)
            if indexFound then
                if countLabelList[indexFound] and countLabelList[indexFound].Parent then
                    argsList[indexFound] = eventData.args
                    countLabelList[indexFound].Text = tostring(tonumber(countLabelList[indexFound].Text) + 1)
                end
            else
                local listButton = Instance.new('TextButton', logScroll)
                listButton.Size = UDim2.new(1, 0, 0, 30)
                listButton.BackgroundColor3 = colorInteractable
                listButton.BorderSizePixel = 0
                listButton.Font = fontRegular
                listButton.Text = ''
                Instance.new('UICorner', listButton).CornerRadius = UDim.new(0, 6)
                listButton.MouseEnter:Connect(function() playTween(listButton, hoverTween, {BackgroundColor3 = colorHover}) end)
                listButton.MouseLeave:Connect(function() playTween(listButton, hoverTween, {BackgroundColor3 = colorInteractable}) end)
                
                local countLabel = Instance.new('TextLabel', listButton)
                countLabel.Size = UDim2.fromOffset(30, 30)
                countLabel.Position = UDim2.fromOffset(5, 0)
                countLabel.BackgroundTransparency = 1
                countLabel.Font = fontBold
                countLabel.Text = '1'
                countLabel.TextColor3 = textColorDark
                countLabel.TextSize = 14
                countLabel.TextXAlignment = alignLeft
                
                local nameLabel = Instance.new('TextLabel', listButton)
                nameLabel.Size = UDim2.new(1, -35, 1, 0)
                nameLabel.Position = UDim2.fromOffset(35, 0)
                nameLabel.BackgroundTransparency = 1
                nameLabel.Font = fontRegular
                nameLabel.Text = eventData.remote.Name
                nameLabel.TextColor3 = textColorNormal
                nameLabel.TextSize = 14
                nameLabel.TextXAlignment = alignLeft
                
                local newIndex = #remoteList + 1
                remoteList[newIndex] = eventData.remote
                argsList[newIndex] = eventData.args
                countLabelList[newIndex] = countLabel
                scriptList[newIndex] = eventData.script
                listElements[newIndex] = listButton
                
                local currentQuery = searchBox.Text:lower()
                if currentQuery ~= '' then 
                    listButton.Visible = eventData.remote.Name:lower():find(currentQuery, 1, true) ~= nil 
                end
                listButton.MouseButton1Click:Connect(function() openInfoPanel(eventData.remote, argsList[newIndex]) end)
            end
        end
    end

    local oldNamecall
    do
        local metaTable = getmetatable(game)

    end

    local function hookNamecall()
        if spyHooked then return end
        if not hookmetamethod or not getnamecallmethod then return end
        if getmetatable(game) then
            oldNamecall = hookmetamethod(game, '__namecall', function(self, ...)
                if not (checkcaller and checkcaller()) then
                    local methodName = getnamecallmethod()
                    if methodName == 'FireServer' and self.ClassName == 'RemoteEvent' then
                        captureRemote(true, self, ...)
                    elseif methodName == 'InvokeServer' and self.ClassName == 'RemoteFunction' then
                        captureRemote(false, self, ...)
                    end
                end
                return oldNamecall(self, ...)
            end)
            spyHooked = true
        end
    end

    local function unhookNamecall()
        if not spyHooked or not oldNamecall then return end
        if not hookmetamethod then return end
        if getmetatable(game) then
            hookmetamethod(game, '__namecall', oldNamecall)
            spyHooked = false
        end
    end

    spyToggleEvent:Connect(function(isOn) 
        spyActive = isOn
        if isOn then hookNamecall() else unhookNamecall() end 
    end)
